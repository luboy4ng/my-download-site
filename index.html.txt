  <script>
    const STORAGE_KEY = 'simple_download_site_items';
    const SITE_KEY = 'simple_download_site_config';

    const defaultItems = [
      {
        id: Date.now() + 1,
        title: '示例工具包',
        type: '工具',
        link: 'https://example.com/tool.zip',
        desc: '这里可以放你的工具整合包下载地址。',
        time: new Date().toLocaleString()
      },
      {
        id: Date.now() + 2,
        title: '示例地图资源',
        type: '地图',
        link: 'https://example.com/map.w3x',
        desc: '这里可以放地图文件或相关资源链接。',
        time: new Date().toLocaleString()
      }
    ];

    const defaultSite = {
      name: 'Lu Boyan 文件下载站',
      desc: '这里集中整理我的文档、工具、地图资源和压缩包。你可以直接点击下载，也可以用搜索快速找到需要的文件。'
    };

    const el = id => document.getElementById(id);

    function escapeHtml(text) {
      return String(text || '')
        .replace(/&/g, '&amp;')
        .replace(/</g, '&lt;')
        .replace(/>/g, '&gt;')
        .replace(/"/g, '&quot;')
        .replace(/'/g, '&#039;');
    }

    function getItems() {
      const raw = localStorage.getItem(STORAGE_KEY);
      return raw ? JSON.parse(raw) : defaultItems;
    }

    function saveItems(items) {
      localStorage.setItem(STORAGE_KEY, JSON.stringify(items));
    }

    function getSite() {
      const raw = localStorage.getItem(SITE_KEY);
      return raw ? JSON.parse(raw) : defaultSite;
    }

    function saveSite(site) {
      localStorage.setItem(SITE_KEY, JSON.stringify(site));
    }

    function renderSite() {
      const site = getSite();
      el('siteTitle').textContent = site.name;
      el('siteDesc').textContent = site.desc;
      el('siteNameInput').value = site.name;
      el('siteDescInput').value = site.desc;
      document.title = site.name;
    }

    function renderList() {
      const keyword = el('searchInput').value.trim().toLowerCase();
      const typeFilter = el('filterInput').value;
      const items = getItems().filter(item => {
        const hitKeyword = item.title.toLowerCase().includes(keyword) || item.desc.toLowerCase().includes(keyword);
        const hitType = typeFilter === '全部' || item.type === typeFilter;
        return hitKeyword && hitType;
      });

      if (!items.length) {
        el('list').innerHTML = '<div class="empty">还没有资源，先在左侧添加一个下载项目吧。</div>';
        return;
      }

      el('list').innerHTML = items.map(item => `
        <div class="item">
          <div class="meta">
            <div class="tag">${escapeHtml(item.type)}</div>
            <h3>${escapeHtml(item.title)}</h3>
            <p>${escapeHtml(item.desc || '暂无说明')}</p>
            <div class="sub">添加时间：${escapeHtml(item.time || '')}</div>
          </div>
          <div class="actions">
            <a class="download" href="${item.link}" target="_blank" rel="noopener noreferrer">下载</a>
            <button class="remove" onclick="removeItem(${item.id})">删除</button>
          </div>
        </div>
      `).join('');
    }

    function addItem() {
      const title = el('fileTitle').value.trim();
      const type = el('fileType').value;
      const link = el('fileLink').value.trim();
      const desc = el('fileDesc').value.trim();

      if (!title || !link) {
        alert('请填写资源标题和下载链接');
        return;
      }

      const items = getItems();
      items.unshift({
        id: Date.now(),
        title,
        type,
        link,
        desc,
        time: new Date().toLocaleString()
      });
      saveItems(items);

      el('fileTitle').value = '';
      el('fileType').value = '工具';
      el('fileLink').value = '';
      el('fileDesc').value = '';
      renderList();
    }

    function removeItem(id) {
      const items = getItems().filter(item => item.id !== id);
      saveItems(items);
      renderList();
    }

    function updateSiteConfig() {
      saveSite({
        name: el('siteNameInput').value.trim() || defaultSite.name,
        desc: el('siteDescInput').value.trim() || defaultSite.desc
      });
      renderSite();
    }

    el('addBtn').addEventListener('click', addItem);
    el('searchInput').addEventListener('input', renderList);
    el('filterInput').addEventListener('change', renderList);
    el('siteNameInput').addEventListener('input', updateSiteConfig);
    el('siteDescInput').addEventListener('input', updateSiteConfig);

    window.removeItem = removeItem;
    renderSite();
    renderList();
  </script>
